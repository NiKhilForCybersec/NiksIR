Using These Queries

These queries are starting points for threat hunting. Adjust time ranges, thresholds, and exclusions for your environment. Review results carefullyâ€” not all matches are malicious.

Persistence Hunting

Hunt for mechanisms attackers use to maintain access across reboots.

KQL T1547.001 - Registry Run Keys
DeviceRegistryEvents | where TimeGenerated > ago(7d) | where ActionType == "RegistryValueSet" | where RegistryKey has_any ( @"\CurrentVersion\Run", @"\CurrentVersion\RunOnce", @"\Explorer\Shell Folders", @"\Explorer\User Shell Folders" ) | where RegistryValueData has_any ( ".exe", ".dll", ".bat", ".ps1", ".vbs", "powershell", "cmd", "wscript", "mshta" ) | project TimeGenerated, DeviceName, RegistryKey, RegistryValueName, RegistryValueData, InitiatingProcessFileName, InitiatingProcessAccountName
KQL T1053.005 - Scheduled Tasks Created via Command Line
DeviceProcessEvents | where TimeGenerated > ago(7d) | where FileName =~ "schtasks.exe" | where ProcessCommandLine has "/create" | extend TaskName = extract(@'/tn\s+["\']?([^"\'\/]+)', 1, ProcessCommandLine) | extend TaskAction = extract(@'/tr\s+["\']?([^"\']+)', 1, ProcessCommandLine) | project TimeGenerated, DeviceName, AccountName, TaskName, TaskAction, ProcessCommandLine, InitiatingProcessFileName
KQL T1543.003 - New Services with Suspicious Paths
SecurityEvent | where TimeGenerated > ago(7d) | where EventID == 7045 | extend ServiceName = tostring(EventData.ServiceName) | extend ServicePath = tostring(EventData.ImagePath) | where ServicePath has_any ("temp", "tmp", "appdata", "public", "programdata") or ServicePath has_any ("cmd", "powershell", "wscript", "mshta") or ServicePath matches regex @"^[A-Za-z]:\\[^\\]+\.exe$" // Root of drive | project TimeGenerated, Computer, ServiceName, ServicePath, SubjectUserName
KQL T1546.003 - WMI Event Subscription
DeviceEvents | where TimeGenerated > ago(7d) | where ActionType == "WmiBindEventFilter" or ActionType has "Wmi" | project TimeGenerated, DeviceName, ActionType, AdditionalFields, InitiatingProcessFileName, InitiatingProcessAccountName

Execution Hunting

Hunt for suspicious code execution patterns.

KQL T1059.001 - Encoded PowerShell Commands
DeviceProcessEvents | where TimeGenerated > ago(7d) | where FileName =~ "powershell.exe" or FileName =~ "pwsh.exe" | where ProcessCommandLine has_any ("-enc", "-encodedcommand", "-e ") | extend EncodedCmd = extract(@"-e[ncodedcommand]*\s+([A-Za-z0-9+/=]+)", 1, ProcessCommandLine) | where strlen(EncodedCmd) > 50 // Filter short/empty | project TimeGenerated, DeviceName, AccountName, ProcessCommandLine, EncodedCmd, InitiatingProcessFileName, InitiatingProcessCommandLine
KQL T1204.002 - Office Spawning Suspicious Processes
let OfficeApps = dynamic(["WINWORD.EXE", "EXCEL.EXE", "POWERPNT.EXE", "OUTLOOK.EXE", "MSACCESS.EXE"]); let SuspiciousChildren = dynamic(["cmd.exe", "powershell.exe", "pwsh.exe", "wscript.exe", "cscript.exe", "mshta.exe", "regsvr32.exe", "rundll32.exe"]); DeviceProcessEvents | where TimeGenerated > ago(7d) | where InitiatingProcessFileName in~ (OfficeApps) | where FileName in~ (SuspiciousChildren) | project TimeGenerated, DeviceName, AccountName, ParentProcess = InitiatingProcessFileName, ChildProcess = FileName, ProcessCommandLine
KQL T1218 - LOLBin Execution with Network Activity
let LOLBins = dynamic(["certutil.exe", "mshta.exe", "regsvr32.exe", "rundll32.exe", "msiexec.exe", "bitsadmin.exe"]); DeviceNetworkEvents | where TimeGenerated > ago(7d) | where InitiatingProcessFileName in~ (LOLBins) | where RemoteIPType == "Public" | summarize ConnectionCount = count(), RemoteIPs = make_set(RemoteIP, 10), URLs = make_set(RemoteUrl, 10) by DeviceName, InitiatingProcessFileName, InitiatingProcessCommandLine, AccountName | order by ConnectionCount desc

Credential Access Hunting

Hunt for credential theft and harvesting attempts.

KQL T1003.001 - LSASS Access
DeviceEvents | where TimeGenerated > ago(7d) | where ActionType == "ProcessAccessed" | where FileName =~ "lsass.exe" | where InitiatingProcessFileName !in~ ( "MsMpEng.exe", "csrss.exe", "svchost.exe", "wininit.exe", "lsass.exe", "services.exe" ) | project TimeGenerated, DeviceName, InitiatingProcessFileName, InitiatingProcessCommandLine, InitiatingProcessAccountName
KQL T1558.003 - Kerberoasting (RC4 Service Ticket Requests)
SecurityEvent | where TimeGenerated > ago(7d) | where EventID == 4769 | where TicketEncryptionType == "0x17" // RC4 | where ServiceName !endswith "$" // Exclude machine accounts | where ServiceName != "krbtgt" | summarize RequestCount = count(), TargetServices = make_set(ServiceName, 20), ServiceCount = dcount(ServiceName) by IpAddress, TargetUserName, bin(TimeGenerated, 1h) | where ServiceCount > 5 // Multiple services = suspicious | order by ServiceCount desc
KQL T1110.003 - Password Spraying
SigninLogs | where TimeGenerated > ago(24h) | where ResultType == "50126" // Invalid password | summarize FailedAccounts = dcount(UserPrincipalName), AccountList = make_set(UserPrincipalName, 50), FailureCount = count() by IPAddress, bin(TimeGenerated, 30m) | where FailedAccounts > 10 // Many accounts from same IP | order by FailedAccounts desc

Lateral Movement Hunting

Hunt for attackers moving through the network.

KQL T1021.002 - SMB/Admin Share Access
SecurityEvent | where TimeGenerated > ago(7d) | where EventID == 5140 // Network share accessed | where ShareName in ("\\\\*\\ADMIN$", "\\\\*\\C$", "\\\\*\\IPC$") | where SubjectUserName !endswith "$" // Exclude machine accounts | summarize AccessCount = count(), TargetComputers = make_set(Computer, 20) by SubjectUserName, IpAddress, ShareName | where AccessCount > 3 | order by AccessCount desc
KQL T1021.001 - RDP to Multiple Hosts
SecurityEvent | where TimeGenerated > ago(7d) | where EventID == 4624 | where LogonType == 10 // RemoteInteractive (RDP) | summarize RDPCount = count(), TargetHosts = make_set(Computer, 20), HostCount = dcount(Computer) by TargetUserName, IpAddress, bin(TimeGenerated, 1h) | where HostCount > 3 // RDP to multiple hosts | order by HostCount desc
KQL T1047 - Remote WMI Execution
DeviceProcessEvents | where TimeGenerated > ago(7d) | where InitiatingProcessFileName =~ "wmiprvse.exe" | where FileName in~ ("cmd.exe", "powershell.exe", "pwsh.exe") | project TimeGenerated, DeviceName, AccountName, FileName, ProcessCommandLine

Data Exfiltration Hunting

Hunt for data theft indicators.

KQL T1567 - Large Uploads to Cloud Storage
DeviceNetworkEvents | where TimeGenerated > ago(7d) | where RemoteUrl has_any ( "dropbox.com", "drive.google.com", "onedrive.live.com", "mega.nz", "wetransfer.com", "sendspace.com", "file.io", "pastebin.com" ) | summarize TotalBytesSent = sum(SentBytes), ConnectionCount = count() by DeviceName, AccountName, RemoteUrl, bin(TimeGenerated, 1h) | where TotalBytesSent > 100000000 // > 100MB | extend TotalMB = round(TotalBytesSent / 1000000.0, 2) | order by TotalBytesSent desc
KQL T1048 - DNS Tunneling Indicators
DnsEvents | where TimeGenerated > ago(7d) | extend SubdomainLength = strlen(tostring(split(Name, ".")[0])) | extend QueryLength = strlen(Name) | where SubdomainLength > 30 or QueryLength > 60 // Long subdomains | summarize QueryCount = count(), UniqueQueries = dcount(Name), SampleQueries = make_set(Name, 10) by ClientIP, tostring(split(Name, ".")[-2]) // Parent domain | where QueryCount > 50 | order by QueryCount desc
KQL Unusual Outbound Data Volume by Host
DeviceNetworkEvents | where TimeGenerated > ago(24h) | where RemoteIPType == "Public" | summarize TotalSent = sum(SentBytes) by DeviceName | join kind=inner ( DeviceNetworkEvents | where TimeGenerated between (ago(30d) .. ago(1d)) | where RemoteIPType == "Public" | summarize AvgSent = avg(SentBytes) * 24 by DeviceName // Daily avg ) on DeviceName | extend Ratio = TotalSent / AvgSent | where Ratio > 5 // 5x normal | order by Ratio desc

Defense Evasion Hunting

Hunt for attackers trying to avoid detection.

KQL T1562.001 - Security Tools Disabled
DeviceProcessEvents | where TimeGenerated > ago(7d) | where ProcessCommandLine has_any ( "Set-MpPreference", "DisableRealtimeMonitoring", "sc stop", "sc delete", "net stop", "Disable-WindowsOptionalFeature" ) | where ProcessCommandLine has_any ( "defender", "sense", "windefend", "mssense", "securityhealth" ) | project TimeGenerated, DeviceName, AccountName, FileName, ProcessCommandLine, InitiatingProcessFileName
KQL T1070.001 - Event Log Clearing
SecurityEvent | where TimeGenerated > ago(7d) | where EventID == 1102 // Audit log cleared | project TimeGenerated, Computer, SubjectUserName, SubjectDomainName
KQL T1036.005 - Process Name Masquerading
// Processes with common names running from wrong locations let SystemProcesses = datatable(Name:string, ExpectedPath:string) [ "svchost.exe", @"c:\windows\system32\", "lsass.exe", @"c:\windows\system32\", "csrss.exe", @"c:\windows\system32\", "services.exe", @"c:\windows\system32\" ]; DeviceProcessEvents | where TimeGenerated > ago(7d) | join kind=inner SystemProcesses on $left.FileName == $right.Name | where tolower(FolderPath) !startswith ExpectedPath | project TimeGenerated, DeviceName, FileName, FolderPath, ExpectedPath, ProcessCommandLine
