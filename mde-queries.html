MDE Query Examples

L1 L2 Section 4.3 ~15 min read
Query Categories

  • Network Activity

Process Analysis

KQL Process tree for specific host
let targetDevice = "WORKSTATION-01"; DeviceProcessEvents | where TimeGenerated > ago(24h) | where DeviceName =~ targetDevice | project TimeGenerated, FileName, ProcessCommandLine, InitiatingProcessFileName, InitiatingProcessCommandLine, AccountName, ProcessId, InitiatingProcessId | order by TimeGenerated asc
KQL Suspicious PowerShell commands
DeviceProcessEvents | where TimeGenerated > ago(24h) | where FileName in~ ("powershell.exe", "pwsh.exe") | where ProcessCommandLine has_any ( "-enc", "-e ", "-encodedcommand", "downloadstring", "downloadfile", "invoke-webrequest", "iwr", "invoke-expression", "iex", "bypass", "-nop", "hidden", "frombase64", "convertto-securestring" ) | project TimeGenerated, DeviceName, AccountName, ProcessCommandLine, InitiatingProcessFileName | order by TimeGenerated desc
KQL Processes spawned by Office applications
let officeApps = dynamic(["winword.exe", "excel.exe", "powerpnt.exe", "outlook.exe", "msaccess.exe"]); DeviceProcessEvents | where TimeGenerated > ago(7d) | where InitiatingProcessFileName in~ (officeApps) | where FileName !in~ (officeApps) // Exclude Office launching Office | project TimeGenerated, DeviceName, InitiatingProcessFileName, FileName, ProcessCommandLine, AccountName | order by TimeGenerated desc
KQL Unsigned processes in user directories
DeviceProcessEvents | where TimeGenerated > ago(24h) | where FolderPath has_any ("\\Users\\", "\\Temp\\", "\\AppData\\") | where ProcessVersionInfoCompanyName == "" or isempty(ProcessVersionInfoCompanyName) | summarize ExecutionCount = count(), Devices = dcount(DeviceName) by FileName, FolderPath, SHA256 | order by ExecutionCount desc

Network Activity

KQL Outbound connections by process
let targetDevice = "WORKSTATION-01"; DeviceNetworkEvents | where TimeGenerated > ago(24h) | where DeviceName =~ targetDevice | where RemoteIPType == "Public" | summarize ConnectionCount = count(), BytesSent = sum(SentBytes), RemoteIPs = make_set(RemoteIP, 10) by InitiatingProcessFileName, RemotePort | order by ConnectionCount desc
KQL Connections to rare destinations
DeviceNetworkEvents | where TimeGenerated > ago(24h) | where RemoteIPType == "Public" | summarize DeviceCount = dcount(DeviceName), Connections = count() by RemoteIP | where DeviceCount == 1 // Only one device talks to this IP | join kind=inner ( DeviceNetworkEvents | where TimeGenerated > ago(24h) ) on RemoteIP | project DeviceName, RemoteIP, RemotePort, InitiatingProcessFileName, Connections
KQL PowerShell with network connections
DeviceNetworkEvents | where TimeGenerated > ago(24h) | where InitiatingProcessFileName in~ ("powershell.exe", "pwsh.exe") | where RemoteIPType == "Public" | project TimeGenerated, DeviceName, RemoteIP, RemotePort, RemoteUrl, InitiatingProcessCommandLine | order by TimeGenerated desc

File Operations

KQL Files created in temp directories
DeviceFileEvents | where TimeGenerated > ago(24h) | where ActionType == "FileCreated" | where FolderPath has_any ("\\Temp\\", "\\tmp\\", "\\Downloads\\") | where FileName endswith_cs ".exe" or FileName endswith_cs ".dll" or FileName endswith_cs ".ps1" | project TimeGenerated, DeviceName, FileName, FolderPath, InitiatingProcessFileName, SHA256 | order by TimeGenerated desc
KQL File hash lookup across environment
let targetHash = "abc123def456..."; // SHA256 DeviceFileEvents | where TimeGenerated > ago(30d) | where SHA256 == targetHash | project TimeGenerated, DeviceName, ActionType, FileName, FolderPath, InitiatingProcessFileName | order by TimeGenerated asc
KQL Renamed executable files
DeviceFileEvents | where TimeGenerated > ago(7d) | where ActionType == "FileRenamed" | where PreviousFileName endswith ".exe" | where FileName !endswith ".exe" // Renamed to hide extension | project TimeGenerated, DeviceName, PreviousFileName, FileName, FolderPath

Persistence Mechanisms

KQL Registry Run key modifications
DeviceRegistryEvents | where TimeGenerated > ago(7d) | where ActionType in ("RegistryValueSet", "RegistryKeyCreated") | where RegistryKey has_any ( "\\Run", "\\RunOnce", "\\RunServices", "\\RunServicesOnce" ) | project TimeGenerated, DeviceName, RegistryKey, RegistryValueName, RegistryValueData, InitiatingProcessFileName, InitiatingProcessCommandLine | order by TimeGenerated desc
KQL Scheduled task creation
DeviceProcessEvents | where TimeGenerated > ago(7d) | where FileName =~ "schtasks.exe" | where ProcessCommandLine has "/create" | project TimeGenerated, DeviceName, ProcessCommandLine, AccountName, InitiatingProcessFileName | order by TimeGenerated desc
KQL New service installations
DeviceEvents | where TimeGenerated > ago(7d) | where ActionType == "ServiceInstalled" | extend ServiceName = tostring(AdditionalFields.ServiceName) | extend ServicePath = tostring(AdditionalFields.ServicePath) | project TimeGenerated, DeviceName, ServiceName, ServicePath, InitiatingProcessAccountName, InitiatingProcessFileName
KQL Startup folder modifications
DeviceFileEvents | where TimeGenerated > ago(7d) | where FolderPath has "\\Start Menu\\Programs\\Startup" | where ActionType in ("FileCreated", "FileModified") | project TimeGenerated, DeviceName, FileName, FolderPath, InitiatingProcessFileName, SHA256

Credential Access

KQL LSASS access attempts
DeviceProcessEvents | where TimeGenerated > ago(24h) | where FileName =~ "lsass.exe" or ProcessCommandLine has "lsass" | where InitiatingProcessFileName !in~ ("svchost.exe", "services.exe") | project TimeGenerated, DeviceName, InitiatingProcessFileName, InitiatingProcessCommandLine, AccountName
KQL Mimikatz indicators
DeviceProcessEvents | where TimeGenerated > ago(24h) | where ProcessCommandLine has_any ( "sekurlsa", "logonpasswords", "wdigest", "kerberos::list", "crypto::cng", "privilege::debug", "token::elevate" ) | project TimeGenerated, DeviceName, FileName, ProcessCommandLine, AccountName
KQL Credential dumping tools
let credTools = dynamic(["mimikatz", "procdump", "lazagne", "gsecdump", "wce.exe", "pwdump", "cachedump"]); DeviceProcessEvents | where TimeGenerated > ago(7d) | where FileName has_any (credTools) or ProcessCommandLine has_any (credTools) | project TimeGenerated, DeviceName, FileName, ProcessCommandLine, AccountName, SHA256

Threat Hunting Queries

KQL LOLBins execution
let LOLBins = dynamic(["certutil.exe", "mshta.exe", "regsvr32.exe", "rundll32.exe", "msiexec.exe", "wmic.exe", "cscript.exe", "wscript.exe", "bitsadmin.exe", "msconfig.exe"]); DeviceProcessEvents | where TimeGenerated > ago(24h) | where FileName in~ (LOLBins) | where ProcessCommandLine has_any ( "http", "ftp", "/download", "-urlcache", "javascript:", "vbscript:", "/i:" ) | project TimeGenerated, DeviceName, FileName, ProcessCommandLine, AccountName
KQL Rare processes in environment
DeviceProcessEvents | where TimeGenerated > ago(7d) | summarize ExecutionCount = count(), DeviceCount = dcount(DeviceName), Devices = make_set(DeviceName, 5) by FileName | where DeviceCount == 1 and ExecutionCount < 5 | order by ExecutionCount asc
Query Tips
  • Always start with time filter for performance
  • Use =~ for case-insensitive comparison
  • Use has instead of contains for better performance
  • has_any matches any item in the list
  • Save frequently used queries as functions
  • Test queries on small time ranges first
