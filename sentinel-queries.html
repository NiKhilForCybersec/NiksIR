Authentication Queries

KQL Failed logins by user
SigninLogs | where TimeGenerated > ago(24h) | where ResultType != "0" // Non-successful | summarize FailedAttempts = count() by UserPrincipalName, ResultType, ResultDescription, IPAddress | order by FailedAttempts desc | take 50
KQL Successful logins from new locations
let lookback = 30d; let recent = 1d; let historicalLocations = SigninLogs | where TimeGenerated between (ago(lookback) .. ago(recent)) | where ResultType == "0" | distinct UserPrincipalName, Location; SigninLogs | where TimeGenerated > ago(recent) | where ResultType == "0" | join kind=leftanti historicalLocations on UserPrincipalName, Location | project TimeGenerated, UserPrincipalName, Location, IPAddress, AppDisplayName, DeviceDetail
KQL Password spray detection
SigninLogs | where TimeGenerated > ago(1h) | where ResultType == "50126" // Invalid username/password | summarize TargetedAccounts = dcount(UserPrincipalName), AttemptCount = count(), Accounts = make_set(UserPrincipalName, 100) by IPAddress, bin(TimeGenerated, 10m) | where TargetedAccounts > 10 // Many accounts from same IP | order by TargetedAccounts desc
KQL Risky sign-ins
SigninLogs | where TimeGenerated > ago(24h) | where RiskLevelDuringSignIn in ("high", "medium") | project TimeGenerated, UserPrincipalName, RiskLevelDuringSignIn, RiskEventTypes, IPAddress, Location, AppDisplayName, ResultType, ResultDescription | order by TimeGenerated desc

Endpoint/MDE Queries

KQL Suspicious PowerShell execution
DeviceProcessEvents | where TimeGenerated > ago(24h) | where FileName in~ ("powershell.exe", "pwsh.exe") | where ProcessCommandLine has_any ( "-enc", "-encodedcommand", "downloadstring", "invoke-expression", "iex", "bypass", "hidden", "frombase64" ) | project TimeGenerated, DeviceName, AccountName, ProcessCommandLine, InitiatingProcessFileName | order by TimeGenerated desc
KQL Process creation timeline for host
let targetHost = "WORKSTATION-01"; DeviceProcessEvents | where TimeGenerated > ago(24h) | where DeviceName =~ targetHost | project TimeGenerated, FileName, ProcessCommandLine, AccountName, InitiatingProcessFileName, InitiatingProcessCommandLine | order by TimeGenerated asc
KQL Lateral movement indicators (PsExec, WMI)
DeviceProcessEvents | where TimeGenerated > ago(24h) | where // PsExec (FileName =~ "psexec.exe" or FileName =~ "psexesvc.exe") // WMI remote or (FileName =~ "wmic.exe" and ProcessCommandLine contains "/node:") // Remote scheduled task or (FileName =~ "schtasks.exe" and ProcessCommandLine contains "/s ") | project TimeGenerated, DeviceName, FileName, ProcessCommandLine, AccountName
KQL New service installations
DeviceEvents | where TimeGenerated > ago(24h) | where ActionType == "ServiceInstalled" | extend ServiceName = tostring(AdditionalFields.ServiceName) | extend ServicePath = tostring(AdditionalFields.ServicePath) | project TimeGenerated, DeviceName, ServiceName, ServicePath, InitiatingProcessAccountName | order by TimeGenerated desc

Email/Office 365 Queries

KQL Emails with malicious attachments
EmailEvents | where TimeGenerated > ago(24h) | where ThreatTypes has "Malware" | project TimeGenerated, SenderFromAddress, RecipientEmailAddress, Subject, ThreatTypes, DeliveryAction, AttachmentCount, NetworkMessageId | order by TimeGenerated desc
KQL Inbox rule creation
OfficeActivity | where TimeGenerated > ago(7d) | where Operation in ("New-InboxRule", "Set-InboxRule") | extend Parameters = tostring(Parameters) | project TimeGenerated, UserId, Operation, Parameters, ClientIP, ResultStatus | order by TimeGenerated desc
KQL URL clicks from emails
UrlClickEvents | where TimeGenerated > ago(24h) | where ActionType == "ClickBlocked" or ThreatTypes != "" | project TimeGenerated, AccountUpn, Url, ActionType, ThreatTypes, IsClickedThrough, NetworkMessageId | order by TimeGenerated desc
KQL Email forwarding rules
OfficeActivity | where TimeGenerated > ago(7d) | where Operation contains "InboxRule" | where Parameters has_any ("ForwardTo", "RedirectTo", "ForwardAsAttachmentTo") | project TimeGenerated, UserId, Operation, Parameters, ClientIP

Network Queries

KQL Outbound connections to rare destinations
DeviceNetworkEvents | where TimeGenerated > ago(24h) | where RemoteIPType == "Public" | summarize ConnectionCount = count(), Devices = dcount(DeviceName) by RemoteIP | where Devices == 1 and ConnectionCount > 10 // Only one device talks to this IP, but frequently | order by ConnectionCount desc
KQL DNS queries to suspicious domains
DnsEvents | where TimeGenerated > ago(24h) | where Name endswith ".xyz" or Name endswith ".top" or Name endswith ".tk" or strlen(tostring(split(Name, ".")[0])) > 25 // Long subdomain | summarize QueryCount = count() by ClientIP, Name | order by QueryCount desc

Azure Activity Queries

KQL Privileged role assignments
AuditLogs | where TimeGenerated > ago(7d) | where OperationName has "Add member to role" | mv-expand TargetResources | extend RoleName = tostring(TargetResources.displayName) | extend UserAdded = tostring(TargetResources.userPrincipalName) | project TimeGenerated, InitiatedBy.user.userPrincipalName, UserAdded, RoleName, Result | order by TimeGenerated desc
KQL App consent grants
AuditLogs | where TimeGenerated > ago(7d) | where OperationName == "Consent to application" | mv-expand TargetResources | extend AppName = tostring(TargetResources.displayName) | extend Permissions = tostring(TargetResources.modifiedProperties) | project TimeGenerated, InitiatedBy.user.userPrincipalName, AppName, Permissions, Result

Threat Hunting Queries

KQL Living off the land binaries (LOLBins)
let LOLBins = dynamic(["certutil.exe", "mshta.exe", "regsvr32.exe", "rundll32.exe", "msiexec.exe", "wmic.exe", "cscript.exe", "wscript.exe", "bitsadmin.exe"]); DeviceProcessEvents | where TimeGenerated > ago(24h) | where FileName in~ (LOLBins) | where ProcessCommandLine has_any ("http", "ftp", "/download", "-urlcache", "javascript:") | project TimeGenerated, DeviceName, FileName, ProcessCommandLine, AccountName
KQL Persistence via scheduled tasks
DeviceProcessEvents | where TimeGenerated > ago(7d) | where FileName =~ "schtasks.exe" | where ProcessCommandLine has "/create" | project TimeGenerated, DeviceName, ProcessCommandLine, AccountName, InitiatingProcessFileName | order by TimeGenerated desc
Query Tips
  • Start with time filter (where TimeGenerated > ago()) for performance
  • Use =~ for case-insensitive string comparison
  • Use has for substring matching (faster than contains)
  • Limit results with take when exploring data
  • Save commonly used queries as functions
